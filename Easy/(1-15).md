
## 1. List all customers with their first name, last name, and email

```sql
SELECT first_name, last_name, email 
FROM Customers;
```

---

## 2. Find all products with price less than $50

```sql
SELECT product_name, price 
FROM Products 
WHERE price < 50;
```

---

## 3. Show all orders that are in 'Delivered' status

```sql
SELECT order_id, order_date, total_amount 
FROM Orders 
WHERE status = 'Delivered';
```

---

## 4. Display customer names and their phone numbers for customers who joined in January 2024

```sql
SELECT first_name, last_name, phone 
FROM Customers 
WHERE created_at BETWEEN '2024-01-01' AND '2024-01-31';
```

---

## 5. List all categories in alphabetical order

```sql
SELECT category_name 
FROM Categories 
ORDER BY category_name;
```

---

## 6. Find products with stock less than 30 units

```sql
SELECT product_name, stock 
FROM Products 
WHERE stock < 30;
```

---

## 7. Show orders placed between January 15, 2024 and February 15, 2024

```sql
SELECT order_id, order_date, total_amount 
FROM Orders 
WHERE order_date BETWEEN '2024-01-15' AND '2024-02-15';
```

---

## 8. Display distinct cities where customers live

```sql
SELECT DISTINCT city 
FROM Addresses 
ORDER BY city;
```

---

## 9. Find customers whose last name starts with 'M'

```sql
SELECT first_name, last_name, email 
FROM Customers 
WHERE last_name LIKE 'M%';
```

---

## 10. List products with price between $50 and $200

```sql
SELECT product_name, price 
FROM Products 
WHERE price BETWEEN 50 AND 200;
```

---

## 11. Show the 5 most expensive products

```sql
SELECT product_name, price 
FROM Products 
ORDER BY price DESC 
LIMIT 5;
```

---

## 12. Count total number of customers

```sql
SELECT COUNT(*) AS total_customers 
FROM Customers;
```

---

## 13. Find the average price of all products

```sql
SELECT ROUND(AVG(price), 2) AS average_price 
FROM Products;
```

---

## 14. Display orders that are not yet shipped (Pending or Processing)

```sql
SELECT order_id, status, order_date 
FROM Orders 
WHERE status IN ('Pending', 'Processing');
```

---

## 15. List customers who don't have a phone number recorded

```sql
SELECT first_name, last_name, email 
FROM Customers 
WHERE phone IS NULL OR phone = '';
```
